          args: {
              transactionHash: string;
              type?: TransactionQueryType;
              exclude_uncommitted?: boolean;
          },
      ): Promise<TransactionResponse>

      Queries on-chain transactions by their transaction hash, returning both pending and committed transactions.

      Type Parameters

      • T extends TransactionResponse

        The type of the transaction to be returned.

      Parameters

      • args: {
            transactionHash: string;
            type?: TransactionQueryType;
            exclude_uncommitted?: boolean;
        }

        The arguments for querying the transaction.

        • transactionHash: string

          The transaction hash should be a hex-encoded bytes string with a 0x prefix.

        • Optionaltype?: TransactionQueryType

          The type of transaction to query.

        • Optionalexclude_uncommitted?: boolean

          Whether to exclude uncommitted transactions.

      Returns Promise<TransactionResponse>

      A Promise that resolves to a TransactionResponse object.

      Example

      import {SupraClient, Network} from "supra-l1-sdk";

      const supra = new SupraClient({ network: Network.TESTNET });

      async function runExample() {
        // Fetch a transaction by its hash
        const transaction = await supra.getTransactionByHash({ transactionHash: "0x123" }); // replace with a real transaction hash

        console.log(transaction);
      }
      runExample().catch(console.error);
